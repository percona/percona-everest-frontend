import { ThemeContextProvider } from './theme-context-provider';

A wrapper that creates a context that allows to toggle light/dark mode.

### Component usage

```js
import {
  ColorModeContext,
  ThemeContextProvider,
} from '@percona/design.theme-context-provider';
import { baseThemeOptions } from '@percona/design.themes.base';

const Toggler = () => {
  const colorMode = React.useContext(ColorModeContext);

  return <button onClick={colorMode.toggleColorMode}>Toggle</button>;
};

<ThemeContextProvider themeOptions={baseThemeOptions}>
  <Toggler />
  <RestOfTheApp />
</ThemeContextProvider>;
```
