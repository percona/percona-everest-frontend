---
description: A custom toggle card that has the same behaviour as MUI's ToggleButton.
---

### Component usage

```js
// stepper with noConnector
<Stepper activeStep={activeStep} noConnector>
   {steps.map(({ label }, index) => {
      const stepProps: { completed?: boolean } = {};
         const labelProps: {
           optional?: React.ReactNode;
         } = {};
         return (
           <Step key={label} {...stepProps}>
             <StepLabel {...labelProps}>{label}</StepLabel>
           </Step>
         );
   })}
</Stepper>

// stepper with connector
<Stepper activeStep={activeStep} >
   {steps.map(({ label }, index) => {
      const stepProps: { completed?: boolean } = {};
         const labelProps: {
           optional?: React.ReactNode;
         } = {};
         return (
           <Step key={label} {...stepProps}>
             <StepLabel {...labelProps}>{label}</StepLabel>
           </Step>
         );
   })}
</Stepper>
```
